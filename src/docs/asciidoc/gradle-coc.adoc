== Convention over configuration

.Before build
----
basic-java
├── build.gradle
└── src
    ├── main
    │   └── java
    │       └── com
    │           └── acme
    │               └── Greeter.java
    └── test
        └── java
            └── com
                └── acme
                    └── GreeterTest.java
----

== Convention over configuration

.After build
----
basic-java/
├── build
│   ├── classes
│   │   ├── main
│   │   │   └── com
│   │   │       └── acme
│   │   │           └── Greeter.class
│   │   └── test
│   │       └── com
│   │           └── acme
│   │               └── GreeterTest.class
│   ├── libs
│   │   └── basic-java.jar
│   ├── reports
│   ├── test-results
│   └── tmp
├── build.gradle
└── src
    ...
----
